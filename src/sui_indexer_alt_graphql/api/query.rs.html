62    /// Fetch a `Node` by its globally unique `ID`. Returns `null` if the node cannot be found (e.g., the underlying data was pruned or never existed).
63    async fn node(&self, ctx: &Context<'_>, id: Id) -> Result<Option<Node>, RpcError> {
64        let scope = self.scope(ctx)?;
65        Ok(match id {
66            Id::Address(a) => Some(Node::Address(Box::new(Address::with_address(scope, a)))),
67
68            Id::Checkpoint(s) => Checkpoint::with_sequence_number(scope, Some(s))
69                .map(Box::new)
70                .map(Node::Checkpoint),
71
72            Id::DynamicFieldByAddress(a) => {
73                let object = Object::with_address(scope, a);
74                DynamicField::from_object(&object, ctx)
75                    .await?
76                    .map(Box::new)
77                    .map(Node::DynamicField)
78            }
79
80            Id::DynamicFieldByRef(a, v, d) => {
81                let object = Object::with_ref(&scope, a, v, d);
82                DynamicField::from_object(&object, ctx)
83                    .await?
84                    .map(Box::new)
85                    .map(Node::DynamicField)
86            }
87
88            Id::Epoch(e) => Some(Node::Epoch(Box::new(Epoch::with_id(scope, e)))),
89
90            Id::MoveObjectByAddress(a) => {
91                let object = Object::with_address(scope, a);
92                MoveObject::from_object(&object, ctx)
93                    .await?
94                    .map(Box::new)
95                    .map(Node::MoveObject)
96            }
97
98            Id::MoveObjectByRef(a, v, d) => {
99                let object = Object::with_ref(&scope, a, v, d);
100                MoveObject::from_object(&object, ctx)
101                    .await?
102                    .map(Box::new)
103                    .map(Node::MoveObject)
104            }
105
106            Id::MovePackage(a) => Some(Node::MovePackage(Box::new(MovePackage::with_address(
107                scope, a,
108            )))),
109
110            Id::ObjectByAddress(a) => Some(Node::Object(Box::new(Object::with_address(scope, a)))),
111
112            Id::ObjectByRef(a, v, d) => {
113                Some(Node::Object(Box::new(Object::with_ref(&scope, a, v, d))))
114            }
115
116            Id::Transaction(d) => Some(Node::Transaction(Box::new(Transaction::with_digest(
117                scope, d,
118            )))),
119        })
120    }

122    /// Look-up an account by its SuiAddress.
123    ///
124    /// If `rootVersion` is specified, nested dynamic field accesses will be fetched at or before this version. This can be used to fetch a child or ancestor object bounded by its root object's version, when its immediate parent is wrapped, or a value in a dynamic object field. For any wrapped or child (object-owned) object, its root object can be defined recursively as:
125    ///
126    /// - The root object of the object it is wrapped in, if it is wrapped.
127    /// - The root object of its owner, if it is owned by another object.
128    /// - The object itself, if it is not object-owned or wrapped.
129    ///
130    /// Specifying a `rootVersion` disables nested queries for paginating owned objects or dynamic fields (these queries are only supported at checkpoint boundaries).
131    async fn address(
132        &self,
133        ctx: &Context<'_>,
134        address: SuiAddress,
135        root_version: Option<UInt53>,
136    ) -> Result<Address, RpcError> {
137        let mut scope = self.scope(ctx)?;
138        if let Some(version) = root_version {
139            scope = scope.with_root_version(version.into());
140        }
141
142        Ok(Address::with_address(scope, address.into()))
143    }

359    /// Fetch an object by its address.
360    ///
361    /// If `version` is specified, the object will be fetched at that exact version.
362    ///
363    /// If `rootVersion` is specified, the object will be fetched at the latest version at or before this version. Nested dynamic field accesses will also be subject to this bound. This can be used to fetch a child or ancestor object bounded by its root object's version. For any wrapped or child (object-owned) object, its root object can be defined recursively as:
364    ///
365    /// - The root object of the object it is wrapped in, if it is wrapped.
366    /// - The root object of its owner, if it is owned by another object.
367    /// - The object itself, if it is not object-owned or wrapped.

385    ) -> Result<Option<Object>, RpcError<object::Error>> {
386        Object::by_key(
387            ctx,
388            self.scope(ctx)?,
389            ObjectKey {
390                address,
391                version,
392                root_version,
393                at_checkpoint,
394            },
395        )
396        .await
397    }

614    /// Types are canonicalized: In the input they can be at any package address at or after the package that first defines them, and in the output they will be relocated to the package that first defines them.
615    ///
616    /// Fails if the type is malformed, returns `null` if a type mentioned does not exist.
617    async fn type_(
618        &self,
619        ctx: &Context<'_>,
620        type_: TypeInput,
621    ) -> Result<Option<MoveType>, RpcError<move_type::Error>> {
622        MoveType::canonicalize(type_.into(), self.scope(ctx)?).await
623    }

625    /// Simulate a transaction to preview its effects without executing it on chain.
626    ///
627    /// Accepts a JSON transaction matching the [Sui gRPC API schema](https://docs.sui.io/references/fullnode-protocol#sui-rpc-v2-Transaction).
628    /// The JSON format allows for partial transaction specification where certain fields can be automatically resolved by the server.
629    ///
630    /// Alternatively, for already serialized transactions, you can pass BCS-encoded data:
631    /// `{"bcs": {"value": "<base64>"}}`
632    ///
633    /// Unlike `executeTransaction`, this does not require signatures since the transaction is not committed to the blockchain. This allows for previewing transaction effects, estimating gas costs, and testing transaction logic without spending gas or requiring valid signatures.
634    ///
635    /// - `checksEnabled`: If true, enables transaction validation checks during simulation. Defaults to true.
636    /// - `doGasSelection`: If true, enables automatic gas coin selection and budget estimation. Defaults to false.
637    async fn simulate_transaction(
638        &self,
639        ctx: &Context<'_>,
640        transaction: Json,
641        checks_enabled: Option<bool>,
642        do_gas_selection: Option<bool>,
643    ) -> Result<SimulationResult, RpcError<TransactionInputError>> {
644        let fullnode_client: &FullnodeClient = ctx.data()?;
645
646        // Convert Json to serde_json::Value and parse as proto::Transaction
647        let json_value: serde_json::Value = transaction
648            .try_into()
649            .map_err(|err| bad_user_input(TransactionInputError::InvalidTransactionJson(err)))?;
650        let proto_tx: proto::Transaction = serde_json::from_value(json_value)
651            .map_err(|err| bad_user_input(TransactionInputError::InvalidTransactionJson(err)))?;
652
653        // Simulate transaction using proto
654        match fullnode_client
655            .simulate_transaction(
656                proto_tx,
657                checks_enabled.unwrap_or(true),
658                do_gas_selection.unwrap_or(false),
659            )
660            .await
661        {
662            Ok(response) => {
663                let scope = self.scope(ctx)?;
664                let tx_data = response
665                    .transaction
666                    .as_ref()
667                    .and_then(|executed_tx| executed_tx.transaction.as_ref())
668                    .and_then(|tx| tx.bcs.as_ref())
669                    .ok_or_else(|| anyhow!("Missing transaction or BCS in simulation response"))?
670                    .deserialize()
671                    .context("Failed to deserialize transaction from response")?;
672
673                SimulationResult::from_simulation_response(scope, response, tx_data).map_err(upcast)
674            }
675            Err(GrpcExecutionError(status)) => Ok(SimulationResult {
676                effects: None,
677                outputs: None,
678                error: Some(status.to_string()),
679            }),
680            Err(other_error) => Err(anyhow!(other_error)
681                .context("Failed to simulate transaction")
682                .into()),
683        }
684    }

686    /// Verify a zkLogin signature os from the given `author`.
687    ///
688    /// Returns a `ZkLoginVerifyResult` where `success` is `true` and `error` is empty if the signature is valid. If the signature is invalid, `success` is `false` and `error` contains the relevant error message.
689    ///
690    /// - `bytes` are either the bytes of a serialized personal message, or `TransactionData`, Base64-encoded.
691    /// - `signature` is a serialized zkLogin signature, also Base64-encoded.
692    /// - `intentScope` indicates whether `bytes` are to be parsed as a personal message or `TransactionData`.
693    /// - `author` is the signer's address.
694    async fn verify_zk_login_signature(
695        &self,
696        ctx: &Context<'_>,
697        bytes: Base64,
698        signature: Base64,
699        intent_scope: ZkLoginIntentScope,
700        author: SuiAddress,
701    ) -> Result<ZkLoginVerifyResult, RpcError<zklogin::Error>> {
702        zklogin::verify_signature(
703            ctx,
704            self.scope(ctx)?,
705            bytes,
706            signature,
707            intent_scope,
708            author,
709        )
710        .await
711    }
